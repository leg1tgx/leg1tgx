// Sample DAX Measure: Year-over-Year Revenue Variance YoY Revenue Variance = VAR CurrentRevenue = CALCULATE(SUM('General Ledger'[Amount]), 'Chart of Accounts'[Class] = "Revenue") VAR PriorRevenue = CALCULATE(SUM('General Ledger'[Amount]), 'Chart of Accounts'[Class] = "Revenue", SAMEPERIODLASTYEAR('Calendar'[Date])) RETURN DIVIDE(CurrentRevenue - PriorRevenue, PriorRevenue, 0)
